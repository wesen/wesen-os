## Workflow Excerpt

```yaml
name: deploy-host-to-k3s

on:
  workflow_dispatch:
    inputs:
      image_ref:
        description: Full image reference to deploy (prefer digest form ghcr.io/...@sha256:...)
        required: true
        type: string

jobs:
  deploy-staging:
    runs-on: ubuntu-latest
    environment: staging
    permissions:
      contents: read
    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          submodules: recursive

      - name: Configure kubeconfig
        env:
          KUBECONFIG_B64: ${{ secrets.KUBECONFIG_B64 }}
        run: |
          test -n "${KUBECONFIG_B64}"
          mkdir -p "$HOME/.kube"
          printf '%s' "${KUBECONFIG_B64}" | base64 --decode > "$HOME/.kube/config"
          chmod 600 "$HOME/.kube/config"

      - name: Verify cluster access
        run: kubectl cluster-info

      - name: Apply base manifests
        run: kubectl apply -k deploy/k8s/wesen-os

      - name: Pin deployment image
        run: |
          kubectl -n wesen-os set image deployment/wesen-os \
            wesen-os="${{ inputs.image_ref }}"

      - name: Wait for rollout
        run: kubectl -n wesen-os rollout status deployment/wesen-os --timeout=180s

      - name: Summarize deployment
        run: |
          echo "environment: staging" >> "$GITHUB_STEP_SUMMARY"
          echo "image_ref: \`${{ inputs.image_ref }}\`" >> "$GITHUB_STEP_SUMMARY"
          echo "" >> "$GITHUB_STEP_SUMMARY"
          kubectl -n wesen-os get deployment wesen-os -o wide >> "$GITHUB_STEP_SUMMARY"
```
